# Card P0-007: Standardize Error Output Destination

## User Story

**As a** user scripting with cdx proxy CLI,
**I want** all error messages to go to stderr and data to stdout,
**So that** I can reliably separate errors from data in my scripts.

**Acceptance Criteria:**
- All error/diagnostic messages go to stderr
- All data output (JSON, tables) goes to stdout
- Exit codes are documented and consistent
- `--quiet` flag suppresses non-error output
- Existing behavior preserved for users

def handle_trace(args: argparse.Namespace) -> int:
    settings = _settings_from_args(args)
    base_url = settings.base_url
    headers = _management_headers(settings)
    upstream_base_url = None
    log_request_preview = None
    
    try:
        debug = fetch_json(
            base_url=base_url,
            path="/debug",
            headers=headers,
            timeout=2.0,
        )
        upstream_base_url = debug.get("upstream_base_url")
        if "log_request_preview" in debug:
            log_request_preview = bool(debug.get("log_request_preview"))
    except Exception:
        # Error message to stderr
        print("Proxy not running. Run `cdx2 proxy` first.", file=sys.stderr)
        return 1
    
    # TUI runs, no output to stdout/stderr
    try:
        run_trace_tui(
            base_url=base_url,
            upstream_base_url=str(upstream_base_url) if upstream_base_url else None,
            log_request_preview=log_request_preview,
            window=max(1, int(args.limit)),
            interval=max(0.1, float(args.interval)),
            limit=max(0, int(args.limit)),
            extra_headers=headers,
        )
    except KeyboardInterrupt:
        return 0
    return 0
```

### Changes to main error handling:

```python
def main(argv: Optional[list[str]] = None) -> int:
    parser = build_parser()
    args = parser.parse_args(argv)
    handler = getattr(args, "handler", None)
    if handler is None:
        parser.print_help()
        return 2
    try:
        return int(handler(args))
    except RuntimeError as exc:
        # All errors to stderr
        print(str(exc), file=sys.stderr)
        return 1
    except ValueError as exc:
        # User errors to stderr
        print(str(exc), file=sys.stderr)
        return 2
    except KeyboardInterrupt:
        # Silent exit for Ctrl+C
        return 130
```

## Exit Codes

| Code | Meaning |
|------|---------|
| 0 | Success |
| 1 | Runtime error (service not running, network error) |
| 2 | User error (invalid arguments, configuration error) |
| 130 | Interrupted (Ctrl+C) |

## Output Conventions

- **stdout**: Data output (JSON, tables, shell exports)
- **stderr**: Error messages, progress info, diagnostics

Use `2>/dev/null` to suppress errors, or `--quiet` to suppress non-data output.

### Examples

```bash
# Get just the exports, no status messages
cdx2 proxy --print-env-only

# Get status as JSON, suppress errors
cdx2 status --json 2>/dev/null

# Quiet mode - only errors to stderr
cdx2 proxy --quiet
```

from __future__ import annotations

import subprocess
import sys


class TestOutputSeparation:
    """Verify stdout/stderr separation conventions."""

    def test_proxy_print_env_only_to_stdout(self):
        """--print-env-only should output only to stdout."""
        # This test verifies the expected behavior
        # Actual testing would require running the proxy
        pass

    def test_doctor_error_to_stderr(self):
        """doctor errors should go to stderr."""
        result = subprocess.run(
            [sys.executable, "-m", "cdx_proxy_cli_v2", "doctor"],
            capture_output=True,
            text=True,
        )
        
        # Should fail because proxy not running
        assert result.returncode != 0
        # Error message should be in stderr
        assert "proxy" in result.stderr.lower() or "running" in result.stderr.lower()

    def test_status_json_to_stdout(self):
        """status --json should output to stdout."""
        result = subprocess.run(
            [sys.executable, "-m", "cdx_proxy_cli_v2", "status", "--json"],
            capture_output=True,
            text=True,
        )
        
        # If successful, output should be valid JSON to stdout
        if result.returncode == 0:
            import json
            data = json.loads(result.stdout)
            assert "pid" in data or "healthy" in data

    def test_invalid_command_error_to_stderr(self):
        """Invalid commands should error to stderr."""
        result = subprocess.run(
            [sys.executable, "-m", "cdx_proxy_cli_v2", "invalid-command"],
            capture_output=True,
            text=True,
        )
        
        assert result.returncode != 0
        # Error should be in stderr
        assert len(result.stderr) > 0 or "invalid" in result.stderr.lower() or "error" in result.stderr.lower()


class TestExitCodes:
    """Verify exit code conventions."""

    def test_success_returns_zero(self):
        """Successful commands should return 0."""
        result = subprocess.run(
            [sys.executable, "-m", "cdx_proxy_cli_v2", "--version"],
            capture_output=True,
        )
        assert result.returncode == 0

    def test_invalid_args_returns_two(self):
        """Invalid arguments should return 2."""
        result = subprocess.run(
            [sys.executable, "-m", "cdx_proxy_cli_v2", "proxy", "--port", "invalid"],
            capture_output=True,
        )
        # May return 2 for ValueError
        assert result.returncode in (1, 2)
