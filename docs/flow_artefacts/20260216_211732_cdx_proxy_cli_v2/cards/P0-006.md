```python
"""Proxy runtime state and coordination."""
from __future__ import annotations

import time
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

from cdx_proxy_cli_v2.auth.rotation import RoundRobinAuthPool
from cdx_proxy_cli_v2.config.settings import Settings
from cdx_proxy_cli_v2.observability.event_log import EventLogger
from cdx_proxy_cli_v2.observability.trace_store import TraceStore


@dataclass
class ProxyRuntime:
    """Coordinates proxy state including auth pool, tracing, and logging."""
    
    settings: Settings
    auth_pool: RoundRobinAuthPool = field(default_factory=RoundRobinAuthPool)
    trace_store: TraceStore = field(init=False)
    logger: EventLogger = field(init=False)

    def __post_init__(self) -> None:
        self.trace_store = TraceStore(max_size=self.settings.trace_max)
        self.logger = EventLogger(self.settings.auth_dir)

    def reload_auths(self) -> int:
        """Reload auth records from disk. Returns count of loaded records."""
        from cdx_proxy_cli_v2.auth.store import load_auth_records
        records = load_auth_records(self.settings.auth_dir)
        self.auth_pool.load(records)
        return len(records)

    def health_snapshot(self, *, refresh: bool = False) -> Dict[str, Any]:
        """Get health snapshot of all auth accounts."""
        if refresh:
            self.reload_auths()
        accounts = self.auth_pool.health_snapshot()
        return {
            "ok": bool(accounts),
            "accounts": accounts,
        }

    def trace_events(self, limit: int) -> List[Dict[str, Any]]:
        """Get recent trace events."""
        return self.trace_store.list(limit=limit)

    def debug_payload(self, *, host: str, port: int) -> Dict[str, Any]:
        """Get debug information payload."""
        import os
        return {
            "status": "running",
            "host": host,
            "port": port,
            "base_url": f"http://{host}:{port}",
            "auth_dir": self.settings.auth_dir,
            "auth_count": self.auth_pool.count(),
            "upstream_base_url": self.settings.upstream,
            "log_request_preview": False,
            "management_key_required": bool(self.settings.management_key),
            "trace_max": self.trace_store.max_size,
            "pid": os.getpid(),
            "event_log_file": str(self.logger.path),
        }

    def record_attempt(
        self,
        *,
        request_id: str,
        method: str,
        path: str,
        route: str,
        status: int,
        latency_ms: int,
        auth_name: str,
        auth_email: Optional[str],
        attempt: int,
        client_ip: Optional[str],
        error: Optional[str] = None,
    ) -> None:
        """Record a request attempt in trace store and event log."""
        payload: Dict[str, Any] = {
            "ts": time.time(),
            "request_id": request_id,
            "method": method,
            "path": path,
            "route": route,
            "status": status,
            "latency_ms": latency_ms,
            "auth_file": auth_name,
            "auth_email": auth_email,
            "attempt": attempt,
            "client_ip": client_ip,
        }
        if error:
            payload["error"] = error
        self.trace_store.add(payload)
        self.logger.write(
            level="INFO" if status < 500 else "WARN",
            event="proxy.request",
            message="request attempt completed",
            **payload,
        )
```

```python
"""Management endpoint handlers."""
from __future__ import annotations

import json
from typing import Any, Dict
from urllib.parse import parse_qs, urlsplit


class ManagementHandler:
    """Handles management endpoint requests."""
    
    def __init__(self, runtime, host: str, port: int) -> None:
        self._runtime = runtime
        self._host = host
        self._port = port

    def handle(self, route: str, path: str, send_json_callback) -> bool:
        """Handle a management route. Returns True if handled."""
        if route == "debug":
            send_json_callback(200, self._runtime.debug_payload(
                host=self._host, port=self._port
            ))
            return True
            
        if route == "trace":
            limit = self._parse_limit(path)
            send_json_callback(200, {"events": self._runtime.trace_events(limit=limit)})
            return True
            
        if route == "health":
            refresh = self._parse_refresh(path)
            send_json_callback(200, self._runtime.health_snapshot(refresh=refresh))
            return True
            
        if route == "auth-files":
            send_json_callback(200, {"files": self._runtime.auth_pool.auth_files()})
            return True
            
        if route == "shutdown":
            send_json_callback(200, {"status": "shutting_down"})
            self._runtime.logger.write(
                level="INFO", event="proxy.shutdown_requested", message="shutdown requested"
            )
            return True  # Caller handles actual shutdown
            
        return False

    @staticmethod
    def _parse_limit(path: str) -> int:
        """Parse limit query parameter from path."""
        query = urlsplit(path).query
        if not query:
            return 0
        params = parse_qs(query)
        try:
            return int(params.get("limit", ["0"])[0])
        except (ValueError, TypeError):
            return 0

    @staticmethod
    def _parse_refresh(path: str) -> bool:
        """Parse refresh query parameter from path."""
        query = urlsplit(path).query
        if not query:
            return False
        params = parse_qs(query)
        return params.get("refresh", ["0"])[0] == "1"


def extract_error_code(raw_body: bytes) -> str | None:
    """Extract error code from JSON response body."""
    if not raw_body:
        return None
    try:
        parsed = json.loads(raw_body.decode("utf-8", errors="replace"))
    except Exception:
        return None
    if isinstance(parsed, dict):
        error = parsed.get("error")
        if isinstance(error, dict):
            code = error.get("code")
            if isinstance(code, str) and code.strip():
                return code.strip()
        code = parsed.get("code")
        if isinstance(code, str) and code.strip():
            return code.strip()
    return None
```

```python
"""SSE streaming response handling."""
from __future__ import annotations

import http.client
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from http.client import HTTPResponse


def is_streaming_response(content_type: str) -> bool:
    """Check if response is a streaming SSE response."""
    return "text/event-stream" in content_type.lower()


def stream_response(
    response: HTTPResponse,
    writer,
    chunk_size: int = 8192,
) -> None:
    """Stream response body to writer.
    
    Args:
        response: HTTP response to stream from
        writer: Object with write() method (typically handler.wfile)
        chunk_size: Size of chunks to read/write
    """
    try:
        while True:
            chunk = response.read(chunk_size)
            if not chunk:
                break
            writer.write(chunk)
            writer.flush()
    finally:
        try:
            response.close()
        except Exception:
            pass


def close_connection(connection: http.client.HTTPConnection | None) -> None:
    """Safely close an HTTP connection."""
    if connection is not None:
        try:
            connection.close()
        except Exception:
            pass
```

```python
from __future__ import annotations

import hmac
import http.client
import json
import time
import uuid
from http.server import BaseHTTPRequestHandler
from typing import Any, Dict, List, Optional, Tuple
from urllib.parse import urlsplit

from cdx_proxy_cli_v2.proxy.rules import (

)
from cdx_proxy_cli_v2.proxy.streaming import (

    is_streaming_response,
    stream_response,
)

from cdx_proxy_cli_v2.proxy.server import DEFAULT_MAX_REQUEST_BODY, DEFAULT_MAX_RESPONSE_BODY


class ProxyHandler(BaseHTTPRequestHandler):
    """HTTP request handler with management endpoints and proxying."""
    
    server: "ProxyHTTPServer"

    def log_message(self, _format: str, *_args: object) -> None:
        """Suppress default logging."""
        return

    def do_GET(self) -> None:
        self._handle_request()

    def do_POST(self) -> None:
        self._handle_request()

    def do_PUT(self) -> None:
        self._handle_request()

    def do_PATCH(self) -> None:
        self._handle_request()

    def do_DELETE(self) -> None:
        self._handle_request()

    def _handle_request(self) -> None:
        """Route request to management or proxy handler."""
        route = management_route(self.path)
        if route:
            if not self._authorize_management():
                self._send_json(401, {"error": "unauthorized management request"})
                return
            self._handle_management(route)
            return
        self._proxy_request()

    def _authorize_management(self) -> bool:
        """Validate management key using constant-time comparison."""
        expected = str(self.server.runtime.settings.management_key or "")
        if not expected:
            return True
        provided = str(self.headers.get("X-Management-Key") or "")
        return hmac.compare_digest(provided, expected)

    def _send_json(self, status: int, payload: Dict[str, Any]) -> None:
        """Send JSON response."""
        raw = json.dumps(payload).encode("utf-8")
        self.send_response(status)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(raw)))
        self.end_headers()
        self.wfile.write(raw)

    def _read_body(self) -> Optional[bytes]:
        """Read request body with size validation."""
        raw_length = self.headers.get("Content-Length", "0")
        try:
            length = int(raw_length)
        except (ValueError, OverflowError):
            self._send_json(400, {"error": "invalid content length"})
            return None
        if length < 0:
            self._send_json(400, {"error": "invalid content length"})
            return None
        if length == 0:
            return b""
        if length > DEFAULT_MAX_REQUEST_BODY:
            self._send_json(413, {"error": "request body too large"})
            return None
        return self.rfile.read(length)

    def _handle_management(self, route: str) -> None:
        """Handle management endpoint request."""
        host, port = self.server.server_address[:2]
        handler = ManagementHandler(self.server.runtime, host, port)
        
        if handler.handle(route, self.path, self._send_json):
            if route == "shutdown":
                self.server.initiate_shutdown()
            return
            
        self._send_json(404, {"error": "unknown management route"})

    def _proxy_request(self) -> None:
        """Proxy request to upstream server with auth rotation."""
        # Implementation delegated to _execute_proxy for clarity
        runtime = self.server.runtime
        upstream = urlsplit(runtime.settings.upstream)
        scheme = upstream.scheme or "https"
        host = upstream.hostname
        
        if not host:
            self._send_json(500, {"error": "invalid upstream host"})
            return
            
        port = upstream.port or (443 if scheme == "https" else 80)
        base_path = upstream.path.rstrip("/")
        
        incoming_path = self.path if self.path.startswith("/") else f"/{self.path}"
        route = trace_route(incoming_path)
        rewritten_path = rewrite_request_path(
            req_path=incoming_path,
            upstream_host=host,
            upstream_base_path=base_path,
        )
        
        if base_path and rewritten_path.startswith(f"{base_path}/"):
            full_path = rewritten_path
        else:
            full_path = f"{base_path}{rewritten_path}" if base_path else rewritten_path
        
        body = self._read_body()
        if body is None:
            return
            
        self._execute_upstream_request(
            scheme=scheme,
            host=host,
            port=port,
            full_path=full_path,
            body=body,
            route=route,
        )
```

```python
"""HTTP proxy server entry point."""
from __future__ import annotations

import argparse
import signal
import threading
from http.server import ThreadingHTTPServer
from typing import Any, Dict, List, Optional, Tuple

from cdx_proxy_cli_v2.config.settings import Settings, build_settings
from cdx_proxy_cli_v2.proxy.rules import is_loopback_host
from cdx_proxy_cli_v2.proxy.runtime import ProxyRuntime
from cdx_proxy_cli_v2.proxy.handler import ProxyHandler

DEFAULT_MAX_REQUEST_BODY = 10 * 1024 * 1024
DEFAULT_MAX_RESPONSE_BODY = 10 * 1024 * 1024


class ProxyHTTPServer(ThreadingHTTPServer):
    """Threaded HTTP server with proxy runtime."""
    
    def __init__(self, server_address: Tuple[str, int], runtime: ProxyRuntime):
        super().__init__(server_address, ProxyHandler)
        self.runtime = runtime

    def initiate_shutdown(self) -> None:
        """Initiate graceful shutdown in background thread."""
        threading.Thread(target=self.shutdown, daemon=True).start()


def run_proxy_server(settings: Settings) -> None:
    """Start and run the proxy server."""
    if not is_loopback_host(settings.host) and not settings.allow_non_loopback:
        raise ValueError("non-loopback bind blocked; use --allow-non-loopback to override")
    
    management_key = str(settings.management_key or "").strip()
    if not management_key:
        raise ValueError("management key required")

    runtime = ProxyRuntime(settings=settings.with_management_key(management_key))
    auth_count = runtime.reload_auths()
    if auth_count <= 0:
        raise ValueError(f"no valid auth files found in {settings.auth_dir}")

    server = ProxyHTTPServer((settings.host, settings.port), runtime)
    bound_host, bound_port = server.server_address[:2]
    
    runtime.logger.write(
        level="INFO",
        event="proxy.started",
        message="proxy server started",
        host=bound_host,
        port=bound_port,
        upstream=settings.upstream,
        auth_count=auth_count,
    )

    stop_requested = False

    def _stop(_signum: int, _frame: object) -> None:
        nonlocal stop_requested
        if stop_requested:
            return
        stop_requested = True
        server.initiate_shutdown()

    previous_sigterm = signal.getsignal(signal.SIGTERM)
    previous_sigint = signal.getsignal(signal.SIGINT)
    signal.signal(signal.SIGTERM, _stop)
    signal.signal(signal.SIGINT, _stop)

    try:
        server.serve_forever(poll_interval=0.5)
    finally:
        server.server_close()
        signal.signal(signal.SIGTERM, previous_sigterm)
        signal.signal(signal.SIGINT, previous_sigint)
        runtime.logger.write(
            level="INFO",
            event="proxy.stopped",
            message="proxy server stopped",
            host=bound_host,
            port=bound_port,
        )


def build_parser() -> argparse.ArgumentParser:
    """Build argument parser for server command."""
    parser = argparse.ArgumentParser(description="cdx_proxy_cli_v2 proxy server")
    parser.add_argument("--auth-dir", required=False)
    parser.add_argument("--host", required=False)
    parser.add_argument("--port", type=int, required=False)
    parser.add_argument("--upstream", required=False)
    parser.add_argument("--management-key", required=False)
    parser.add_argument("--trace-max", type=int, required=False)
    parser.add_argument("--allow-non-loopback", action="store_true")
    return parser


def main(argv: Optional[List[str]] = None) -> int:
    """Main entry point for server command."""
    parser = build_parser()
    args = parser.parse_args(argv)
    settings = build_settings(
        auth_dir=args.auth_dir,
        host=args.host,
        port=args.port,
        upstream=args.upstream,
        management_key=args.management_key,
        allow_non_loopback=bool(args.allow_non_loopback),
        trace_max=args.trace_max,
    )
    run_proxy_server(settings)
    return 0
```
