class TestBuildSettingsPrecedence:
    """Tests for build_settings precedence rules."""

    def test_uses_defaults_when_nothing_provided(self, tmp_path: Path, monkeypatch):
        """Should use defaults when no config provided."""
        # Clear any existing env vars
        for key in [ENV_AUTH_DIR, ENV_HOST, ENV_PORT, ENV_UPSTREAM, ENV_MANAGEMENT_KEY]:
            monkeypatch.delenv(key, raising=False)
        
        # Create a temp directory that will be used
        auth_dir = tmp_path / "auths"
        auth_dir.mkdir()
        
        settings = build_settings(auth_dir=str(auth_dir))
        
        assert settings.host == DEFAULT_HOST
        assert settings.upstream == DEFAULT_UPSTREAM
        assert settings.trace_max == DEFAULT_TRACE_MAX

    def test_cli_args_override_env_vars(self, tmp_path: Path, monkeypatch):
        """CLI args should take precedence over env vars."""
        monkeypatch.setenv(ENV_HOST, "env-host")
        
        settings = build_settings(
            auth_dir=str(tmp_path),
            host="cli-host",
        )
        
        assert settings.host == "cli-host"

    def test_env_vars_override_file(self, tmp_path: Path, monkeypatch):
        """Env vars should take precedence over env file."""
        # Create env file
        env_file = tmp_path / ".env"
        env_file.write_text(f"{ENV_HOST}=file-host\n")
        
        # Set env var
        monkeypatch.setenv(ENV_HOST, "env-host")
        monkeypatch.setenv("CLIPROXY_ENV_FILE", str(env_file))
        
        settings = build_settings(auth_dir=str(tmp_path))
        
        assert settings.host == "env-host"

    def test_whitespace_stripped_from_values(self):
        """Should strip whitespace from string values."""
        settings = build_settings(
            auth_dir="/tmp",
            host="  127.0.0.1  ",
            upstream="  https://api.example.com  ",
        )
        
        assert settings.host == "127.0.0.1"
        assert settings.upstream == "https://api.example.com"
